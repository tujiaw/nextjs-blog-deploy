---

在用WTL的时候经常会遇到因为某某头文件没有包含而报错的问题，现将我常用的几个头文件列出来
```

#define _WTL_USE_CSTRING		// 提前声明CString类
#include <atlbase.h>			// 基本的ATL类
#include <atlapp.h>				// 基本的WTL类
extern CAppModule _Module;		// WTL派生的CComModule版本
#include <atlwin.h>				// ATL窗口类
#include <atlddx.h>				// WTL数据交换类
#include <atlframe.h>			//WTL主框架窗口类
#include <atlctrls.h>			//WTL控件类
#include <atlctrlx.h>			
#include <atldlgs.h>			//WTL对话框类
#include <atlctrlw.h>			
#include <atlmisc.h>			// WTL使用工具类，包含：CString
```
